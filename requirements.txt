# ── In-memory job store ────────────────────────────────────────────────────────
jobs: dict[str, dict] = {}
executor = ThreadPoolExecutor(max_workers=2)

CLIP_DIR = Path(tempfile.gettempdir()) / "clipforge_clips"
CLIP_DIR.mkdir(exist_ok=True)

# ── Models ─────────────────────────────────────────────────────────────────────
class ForgeRequest(BaseModel):
    url: str
    formats: list[str] = ["9:16", "16:9", "1:1"]
    max_clips: int = 5
    min_duration: int = 15
    max_duration: int = 60

# ── Helpers ────────────────────────────────────────────────────────────────────
ASPECT_CONFIGS = {
    "9:16":  {"w": 1080, "h": 1920},
    "16:9":  {"w": 1920, "h": 1080},
    "1:1":   {"w": 1080, "h": 1080},
}

def log(job_id: str, msg: str):
    if job_id in jobs:
        jobs[job_id]["log"].append(f"[{datetime.now().strftime('%H:%M:%S')}] {msg}")
        print(f"[{job_id[:8]}] {msg}")

def set_progress(job_id: str, pct: int, stage: str):
    if job_id in jobs:
        jobs[job_id]["progress"] = pct
        jobs[job_id]["stage"] = stage

# ── Video download ─────────────────────────────────────────────────────────────
def download_video(url: str, out_dir: Path, job_id: str) -> Path:
    log(job_id, "Fetching video info…")
    ydl_opts = {
        "format": "bestvideo[ext=mp4][height<=1080]+bestaudio[ext=m4a]/best[ext=mp4]/best",
        "outtmpl": str(out_dir / "source.%(ext)s"),
        "merge_output_format": "mp4",
        "quiet": True,
        "no_warnings": True,
    }
    with yt_dlp.YoutubeDL(ydl_opts) as ydl:
        info = ydl.extract_info(url, download=False)
        duration = info.get("duration", 0)
        if duration < 30:
            raise ValueError("Video too short — minimum 30 seconds required.")
        if duration > 7200:
            raise ValueError("Video too long — maximum 2 hours supported.")
        title = info.get("title", "video")
        log(job_id, f"Downloading: {title} ({duration}s)")
        ydl.download([url])

    mp4 = out_dir / "source.mp4"
    if not mp4.exists():
        candidates = list(out_dir.glob("source.*"))
        if not candidates:
            raise FileNotFoundError("Download failed — no output file found.")
        mp4 = candidates[0]
    return mp4

# ── Audio peak detector ────────────────────────────────────────────────────────
def detect_audio_peaks(video_path: Path, job_id: str) -> list[float]:
    log(job_id, "Analysing audio peaks…")
    audio_path = video_path.parent / "audio.wav"
    os.system(f'ffmpeg -i "{video_path}" -ac 1 -ar 16000 -vn "{audio_path}" -y -loglevel quiet')
    if not audio_path.exists():
        return []
    y, sr = librosa.load(str(audio_path), sr=16000)
    rms = librosa.feature.rms(y=y, frame_length=2048, hop_length=512)[0]
    times = librosa.frames_to_time(np.arange(len(rms)), sr=sr, hop_length=512)
    threshold = np.percentile(rms, 80)
    peaks = []
    for i, (t, r) in enumerate(zip(times, rms)):
        if r > threshold:
            if not peaks or t - peaks[-1] > 10:
                peaks.append(float(t))
    log(job_id, f"Found {len(peaks)} audio peaks")
    return peaks[:20]

# ── Transcript keyword detector ────────────────────────────────────────────────
VIRAL_KEYWORDS = [
    "secret", "never", "always", "most", "best", "worst", "only", "incredible",
    "impossible", "shocking", "surprising", "crazy", "insane", "unbelievable",
    "you need", "you should", "you must", "don't", "stop", "start", "warning",
    "mistake", "hack", "trick", "tip", "fact", "truth", "lie", "exposed",
    "actually", "honestly", "literally", "basically", "seriously",
]

def detect_transcript_moments(video_path: Path, job_id: str) -> list[float]:
    log(job_id, "Transcribing audio (Whisper)…")
    audio_path = video_path.parent / "audio.wav"
    if not audio_path.exists():
        return []
    try:
        model = whisper.load_model("tiny")
        result = model.transcribe(str(audio_path), word_timestamps=True)
        moments = []
        for seg in result.get("segments", []):
            text = seg.get("text", "").lower()
            for kw in VIRAL_KEYWORDS:
                if kw in text:
                    t = float(seg.get("start", 0))
                    if not moments or t - moments[-1] > 10:
                        moments.append(t)
                    break
        log(job_id, f"Found {len(moments)} viral-keyword moments")
        return moments[:20]
    except Exception as e:
        log(job_id, f"Whisper error (skipping): {e}")
        return []

# ── Motion / scene change detector ────────────────────────────────────────────
def detect_motion_moments(video_path: Path, job_id: str) -> list[float]:
    log(job_id, "Scanning for visual action…")
    cap = cv2.VideoCapture(str(video_path))
    fps = cap.get(cv2.CAP_PROP_FPS) or 24
    total = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
    sample_every = max(1, int(fps * 2))  # sample every 2 seconds
    prev_gray = None
    scores = []
    frame_idx = 0
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        if frame_idx % sample_every == 0:
            gray = cv2.cvtColor(cv2.resize(frame, (320, 180)), cv2.COLOR_BGR2GRAY)
            if prev_gray is not None:
                diff = cv2.absdiff(gray, prev_gray)
                score = float(diff.mean())
                t = frame_idx / fps
                scores.append((t, score))
            prev_gray = gray
        frame_idx += 1
    cap.release()

    if not scores:
        return []
    threshold = np.percentile([s for _, s in scores], 85)
    moments = []
    for t, s in scores:
        if s > threshold:
            if not moments or t - moments[-1] > 10:
                moments.append(t)
    log(job_id, f"Found {len(moments)} high-motion moments")
    return moments[:20]

# ── Score & deduplicate moments ────────────────────────────────────────────────
def merge_moments(audio: list[float], transcript: list[float], motion: list[float],
                  video_duration: float, clip_duration: int, max_clips: int) -> list[dict]:
    scored: dict[float, dict] = {}

    def add(t: float, source: str, weight: float):
        bucket = round(t / 5) * 5  # 5-second buckets
        if bucket not in scored:
            scored[bucket] = {"time": t, "score": 0, "sources": []}
        scored[bucket]["score"] += weight
        scored[bucket]["sources"].append(source)

    for t in audio:      add(t, "audio",      1.0)
    for t in transcript: add(t, "transcript", 1.5)
    for t in motion:     add(t, "motion",     0.8)

    candidates = sorted(scored.values(), key=lambda x: -x["score"])
    selected = []
    used_times = []
    for c in candidates:
        t = c["time"]
        # ensure clip fits within video
        end = t + clip_duration
        if end > video_duration:
            t = max(0, video_duration - clip_duration)
        # ensure not overlapping a selected clip
        if any(abs(t - u) < clip_duration * 0.75 for u in used_times):
            continue
        selected.append({"start": t, "score": c["score"], "sources": c["sources"]})
        used_times.append(t)
        if len(selected) >= max_clips:
            break

    # fallback: evenly spaced clips
    if not selected:
        step = video_duration / (max_clips + 1)
        for i in range(1, max_clips + 1):
            t = step * i
            if t + clip_duration <= video_duration:
                selected.append({"start": t, "score": 0, "sources": ["fallback"]})

    return selected

# ── FFmpeg clip cutter ─────────────────────────────────────────────────────────
def cut_clip(video_path: Path, start: float, duration: int, fmt: str,
             out_path: Path, job_id: str) -> bool:
    cfg = ASPECT_CONFIGS[fmt]
    w, h = cfg["w"], cfg["h"]

    # Smart crop: centre-crop to target aspect
    vf = (
        f"scale='if(gt(iw/ih,{w}/{h}),{w},-2)':'if(gt(iw/ih,{w}/{h}),-2,{h})',"
        f"crop={w}:{h}"
    )

    cmd = (
        f'ffmpeg -ss {start:.2f} -i "{video_path}" -t {duration} '
        f'-vf "{vf}" -c:v libx264 -preset veryfast -crf 23 '
        f'-c:a aac -b:a 128k -movflags +faststart '
        f'"{out_path}" -y -loglevel error'
    )
    ret = os.system(cmd)
    return ret == 0 and out_path.exists()

# ── Main processing pipeline ───────────────────────────────────────────────────
def process_job(job_id: str, req: ForgeRequest):
    job = jobs[job_id]
    work_dir = CLIP_DIR / job_id
    work_dir.mkdir(parents=True, exist_ok=True)

    try:
        # 1. Download
        set_progress(job_id, 5, "Downloading video")
        video_path = download_video(req.url, work_dir, job_id)

        # 2. Get duration
        cap = cv2.VideoCapture(str(video_path))
        fps = cap.get(cv2.CAP_PROP_FPS) or 24
        total_frames = cap.get(cv2.CAP_PROP_FRAME_COUNT)
        video_duration = total_frames / fps
        cap.release()
        log(job_id, f"Video duration: {video_duration:.1f}s @ {fps:.1f}fps")

        # 3. Run detectors in parallel
        set_progress(job_id, 20, "Running AI detectors")
        with ThreadPoolExecutor(max_workers=3) as pool:
            f_audio      = pool.submit(detect_audio_peaks,       video_path, job_id)
            f_transcript = pool.submit(detect_transcript_moments, video_path, job_id)
            f_motion     = pool.submit(detect_motion_moments,     video_path, job_id)
            audio_times      = f_audio.result()
            transcript_times = f_transcript.result()
            motion_times     = f_motion.result()

        set_progress(job_id, 50, "Selecting best moments")
        clip_duration = min(req.max_duration, max(req.min_duration, 30))
        moments = merge_moments(
            audio_times, transcript_times, motion_times,
            video_duration, clip_duration, req.max_clips
        )
        log(job_id, f"Selected {len(moments)} moments to clip")

        # 4. Cut clips
        set_progress(job_id, 60, "Cutting clips")
        clips = []
        total_cuts = len(moments) * len(req.formats)
        cut_count = 0

        for i, moment in enumerate(moments):
            for fmt in req.formats:
                safe_fmt = fmt.replace(":", "x")
                fname = f"clip_{i+1}_{safe_fmt}.mp4"
                out_path = work_dir / fname
                ok = cut_clip(video_path, moment["start"], clip_duration, fmt, out_path, job_id)
                if ok:
                    clips.append({
                        "filename": fname,
                        "clip_number": i + 1,
                        "format": fmt,
                        "start": round(moment["start"], 1),
                        "duration": clip_duration,
                        "sources": moment["sources"],
                        "score": round(moment["score"], 2),
                        "download_url": f"/clips/{job_id}/{fname}",
                    })
                    log(job_id, f"✓ clip_{i+1} {fmt}")
                else:
                    log(job_id, f"✗ clip_{i+1} {fmt} — ffmpeg error")
                cut_count += 1
                set_progress(job_id, 60 + int(35 * cut_count / total_cuts), "Cutting clips")

        job["clips"] = clips
        job["status"] = "done"
        set_progress(job_id, 100, "Complete")
        log(job_id, f"Done! {len(clips)} clips ready.")

    except Exception as e:
        job["status"] = "error"
        job["error"] = str(e)
        log(job_id, f"ERROR: {e}")
        traceback.print_exc()

# ── API routes ─────────────────────────────────────────────────────────────────
@app.get("/")
async def health():
    return {"status": "ok", "service": "ClipForge"}

@app.post("/forge")
async def forge(req: ForgeRequest):
    job_id = str(uuid.uuid4())
    jobs[job_id] = {
        "id": job_id,
        "status": "running",
        "progress": 0,
        "stage": "Queued",
        "clips": [],
        "error": None,
        "log": [],
        "created_at": datetime.now().isoformat(),
    }
    executor.submit(process_job, job_id, req)
    return {"job_id": job_id}

@app.get("/status/{job_id}")
async def status(job_id: str):
    job = jobs.get(job_id)
    if not job:
        raise HTTPException(404, "Job not found")
    return {
        "id": job["id"],
        "status": job["status"],
        "progress": job["progress"],
        "stage": job["stage"],
        "clips": job["clips"],
        "error": job["error"],
        "log": job["log"][-20:],
    }

@app.get("/clips/{job_id}/{filename}")
async def download_clip(job_id: str, filename: str):
    path = CLIP_DIR / job_id / filename
    if not path.exists():
        raise HTTPException(404, "Clip not found")
    return FileResponse(
        path,
        media_type="video/mp4",
        headers={"Content-Disposition": f'attachment; filename="{filename}"'},
    )

@app.get("/app", response_class=HTMLResponse)
async def serve_app():
    html_path = Path(__file__).parent.parent / "frontend" / "index.html"
    if html_path.exists():
        return HTMLResponse(html_path.read_text())
    raise HTTPException(404, "Frontend not found")

# ── Cleanup old jobs (>2h) on startup ─────────────────────────────────────────
@app.on_event("startup")
async def cleanup():
    import shutil
    cutoff = datetime.now() - timedelta(hours=2)
    for d in CLIP_DIR.iterdir():
        try:
            if d.is_dir() and datetime.fromtimestamp(d.stat().st_mtime) < cutoff:
                shutil.rmtree(d)
        except:
            pass
